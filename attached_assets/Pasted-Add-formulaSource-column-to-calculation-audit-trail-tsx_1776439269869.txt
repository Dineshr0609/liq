const FORMULA_SOURCE_CONFIG: Record<string, { label: string; color: string; description: string }> = {
  primary_formula_node:              { label: 'Formula',     color: 'bg-green-100 text-green-800',   description: 'Executed from structured formula definition — highest confidence path' },
  rebate_tiered_quarterly_universal: { label: 'Quarterly',   color: 'bg-blue-100 text-blue-800',    description: 'Quarterly rebate aggregation via universal engine — full traceability' },
  table_data_field_mappings:         { label: 'Table',       color: 'bg-purple-100 text-purple-800',description: 'Formula built on-the-fly from AI-extracted table data' },
  base_rate_percentage:              { label: 'Base Rate',   color: 'bg-orange-100 text-orange-800',description: 'AI-extracted percentage rate applied to transaction amount' },
  fixed_amount:                      { label: 'Fixed',       color: 'bg-gray-100 text-gray-700',    description: 'Fixed fee or bonus amount applied once per period' },
  minimum_guarantee:                 { label: 'Min Guar.',   color: 'bg-red-100 text-red-800',      description: 'Minimum guarantee true-up — shortfall between calculated fees and guarantee floor' },
  tiers_array:                       { label: 'Tiers',       color: 'bg-indigo-100 text-indigo-800',description: 'Formula built from AI-extracted tier array in formulaDefinition' },
  volume_tiers_legacy:               { label: 'Legacy',      color: 'bg-yellow-100 text-yellow-800',description: 'Built from legacy volumeTiers field — consider migrating to formulaDefinition' },
  direct_base_rate_final:            { label: 'Fallback',    color: 'bg-amber-100 text-amber-800',  description: 'Final fallback: direct baseRate used — no formula definition found' },
  unknown:                           { label: 'Unknown',     color: 'bg-gray-100 text-gray-400',    description: 'Formula source could not be determined' },
};
 
function FormulaSourceBadge({ source, tierMode }: { source?: string; tierMode?: string }) {
  const cfg = FORMULA_SOURCE_CONFIG[source || 'unknown'] || FORMULA_SOURCE_CONFIG.unknown;
  return (
    <TooltipProvider>
      <Tooltip>
        <TooltipTrigger asChild>
          <span
            className={`inline-flex items-center px-2 py-0.5 rounded text-[10px] font-medium cursor-help ${cfg.color}`}
            data-testid={`badge-formula-source-${source}`}
          >
            {cfg.label}
            {source === 'rebate_tiered_quarterly_universal' && tierMode && (
              <span className={`ml-1 opacity-75`}>
                · {tierMode === 'marginal' ? 'marginal' : 'whole'}
              </span>
            )}
          </span>
        </TooltipTrigger>
        <TooltipContent side="top">
          <p className="text-xs max-w-[220px]">{cfg.description}</p>
        </TooltipContent>
      </Tooltip>
    </TooltipProvider>
  );
}
