Sample Query: List All Contracts with Type and Status

SELECT
  c.id,
  c.contract_number,
  c.display_name,
  ctd.name           AS contract_type_name,
  sf.name            AS subflow_name,
  c.status,
  c.approval_state,
  c.organization_name,
  c.counterparty_name,
  c.effective_start,
  c.effective_end,
  c.current_version,
  c.created_at
FROM contracts c
LEFT JOIN contract_type_definitions ctd
  ON c.contract_type = ctd.code
LEFT JOIN subflows sf
  ON c.contract_subtype = sf.code
ORDER BY c.created_at DESC;

  cm.contract_number,
  comp.name             AS company_name,
  cm.contract_category,

  c.display_name,
  c.contract_number,
  ca.summary,
  ca.confidence,
  ca.processing_time,
  ca.key_terms,
  ca.risk_analysis,
  ca.insights
FROM contract_analysis ca
JOIN contracts c
  ON ca.contract_id = c.id
WHERE ca.contract_id = '<contract_uuid>';

SELECT
  c.display_name,
  ca.confidence,
  ca.header_review_flags
FROM contract_analysis ca
JOIN contracts c ON ca.contract_id = c.id
WHERE ca.confidence < 0.70
ORDER BY ca.confidence ASC;

Sample Query: Partners Assigned to a Contract

SELECT
  pca.association_id,
  pca.partner_id,
  pca.contract_status,
  pca.is_primary_contract,
  pca.effective_date,
  pca.expiration_date,
  c.display_name     AS contract_name
FROM partner_contract_associations pca
JOIN contracts c
  ON pca.contract_id = c.id
WHERE pca.contract_id = '<contract_uuid>'
ORDER BY pca.is_primary_contract DESC, pca.effective_date;

-- All active partner assignments across contracts
SELECT
  pca.partner_id,
  COUNT(DISTINCT pca.contract_id) AS contract_count,
  SUM(CASE WHEN pca.is_primary_contract THEN 1 ELSE 0 END) AS primary_count
FROM partner_contract_associations pca
WHERE pca.contract_status = 'Active'
GROUP BY pca.partner_id;

Sample Query: License Fee Rules for a Contract

SELECT
  rr.rule_name,
  rr.rule_type,
  rr.base_rate,
  rr.minimum_amount,
  rr.maximum_amount,
  rr.product_categories,
  rr.territories,
  rr.confidence,
  rr.source,
  rr.is_active,
  rr.description
FROM royalty_rules rr
WHERE rr.contract_id = '<contract_uuid>'
  AND rr.is_active = true
ORDER BY rr.extraction_order ASC NULLS LAST;

-- Rules by type across all contracts
SELECT
  rr.rule_type,
  COUNT(*) AS rule_count,
  AVG(rr.base_rate) AS avg_base_rate,
  AVG(rr.confidence) AS avg_confidence
FROM royalty_rules rr
WHERE rr.is_active = true
GROUP BY rr.rule_type;

Sample Query: Clauses Grouped by Category

SELECT
  cc.clause_identifier,
  cc.section_ref,
  SUBSTRING(cc.text, 1, 200) AS clause_preview,
  cat.name             AS category_name,
  ft.name              AS flow_type_name,
  cc.affects_accrual,
  cc.confidence
FROM contract_clauses cc
LEFT JOIN clause_categories cat
  ON cc.clause_category_code = cat.code
LEFT JOIN flow_types ft
  ON cc.flow_type_code = ft.code
WHERE cc.contract_id = '<contract_uuid>'
ORDER BY cc.clause_category_code, cc.clause_identifier;

-- Clause count by category for a contract
SELECT
  COALESCE(cat.name, cc.clause_category_code, 'Uncategorized') AS category,
  COUNT(*) AS clause_count,
  ROUND(AVG(cc.confidence)::numeric, 2) AS avg_confidence
FROM contract_clauses cc
LEFT JOIN clause_categories cat ON cc.clause_category_code = cat.code
WHERE cc.contract_id = '<contract_uuid>'
GROUP BY category
ORDER BY clause_count DESC;

Full Contract View Query

Comprehensive query joining contracts + master + analysis into a single row, with counts of related records

Complete Contract Summary (Single Row per Contract)

SELECT
  -- Contract basics
  c.id,
  c.contract_number,
  c.display_name,
  c.status,
  c.approval_state,
  c.current_version,
  c.organization_name,
  c.counterparty_name,
  c.effective_start,
  c.effective_end,

  -- Type info
  ctd.name                  AS contract_type_name,
  sf.name                   AS subflow_name,

  -- Master data
  cm.territory_scope,
  cm.channel_scope,
  cm.payment_frequency,
  cm.currency,
  cm.contract_value_estimated_annual,
  cm.contract_status        AS business_status,
  cm.auto_renew,

  -- AI analysis
  ca.summary               AS ai_summary,
  ca.confidence            AS analysis_confidence,

  -- Counts
  (SELECT COUNT(*) FROM royalty_rules rr
   WHERE rr.contract_id = c.id
     AND rr.is_active = true)     AS active_rules,
  (SELECT COUNT(*) FROM contract_terms ct
   WHERE ct.contract_id = c.id)  AS total_terms,
  (SELECT COUNT(*) FROM contract_clauses cc
   WHERE cc.contract_id = c.id)  AS total_clauses,
  (SELECT COUNT(*) FROM partner_contract_associations pca
   WHERE pca.contract_id = c.id) AS total_partners

FROM contracts c
LEFT JOIN contract_type_definitions ctd
  ON c.contract_type = ctd.code
LEFT JOIN subflows sf
  ON c.contract_subtype = sf.code
LEFT JOIN contract_master cm
  ON cm.linked_contract_id = c.id
LEFT JOIN contract_analysis ca
  ON ca.contract_id = c.id
ORDER BY c.created_at DESC;
