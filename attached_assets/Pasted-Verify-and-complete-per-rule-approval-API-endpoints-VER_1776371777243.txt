Step 2 — Add approve endpoint if missing
If POST /api/contracts/:contractId/rules/:ruleId/approve does not exist, add it to rulesRoutes.ts inside the registerRulesRoutes function:
app.post('/api/contracts/:contractId/rules/:ruleId/approve',
  isAuthenticated,
  async (req: any, res) => {
    const { contractId, ruleId } = req.params;
    const { notes } = req.body;
 
    // Role check — only owner/admin/editor can approve rules
    const contextRole = req.user?.activeContext?.role;
    const canApprove = ['owner', 'admin', 'editor'].includes(contextRole) ||
      req.user?.role === 'admin' || req.user?.isSystemAdmin;
    if (!canApprove) {
      return res.status(403).json({ message: 'Insufficient permissions to approve rules' });
    }
 
    const { contract, authorized, error } = await validateContractAccess(contractId, req.user);
    if (!contract) return res.status(404).json({ message: error || 'Contract not found' });
    if (!authorized) return res.status(403).json({ message: error || 'Access denied' });
 
    await storage.updateRoyaltyRule(ruleId, {
      approvalStatus: 'approved',
      approvedBy: req.user.id,
      approvedAt: new Date(),
      approvalNotes: notes || null,
    });
 
    await createAuditLog(req, 'rule_approved', 'contract_rule', ruleId, { contractId, notes });
    res.json({ message: 'Rule approved successfully' });
  }
);
 
Step 3 — Add reject endpoint if missing
If POST /api/contracts/:contractId/rules/:ruleId/reject does not exist, add it immediately after the approve endpoint:
app.post('/api/contracts/:contractId/rules/:ruleId/reject',
  isAuthenticated,
  async (req: any, res) => {
    const { contractId, ruleId } = req.params;
    const { reason } = req.body;
 
    // Reason is required for reject — Finance must document why
    if (!reason || reason.trim().length === 0) {
      return res.status(400).json({ message: 'Rejection reason is required' });
    }
 
    const contextRole = req.user?.activeContext?.role;
    const canReject = ['owner', 'admin', 'editor'].includes(contextRole) ||
      req.user?.role === 'admin' || req.user?.isSystemAdmin;
    if (!canReject) return res.status(403).json({ message: 'Insufficient permissions' });
 
    const { contract, authorized, error } = await validateContractAccess(contractId, req.user);
    if (!contract) return res.status(404).json({ message: error || 'Contract not found' });
    if (!authorized) return res.status(403).json({ message: error || 'Access denied' });
 
    await storage.updateRoyaltyRule(ruleId, {
      approvalStatus: 'rejected',
      approvedBy: req.user.id,
      approvedAt: new Date(),
      approvalNotes: reason,
    });
 
    await createAuditLog(req, 'rule_rejected', 'contract_rule', ruleId, { contractId, reason });
    res.json({ message: 'Rule rejected' });
  }
);
 
Step 4 — Update HumanReviewQueue.tsx to call these endpoints
In client/src/pages/HumanReviewQueue.tsx, add two useMutation hooks:
const approveMutation = useMutation({
  mutationFn: ({ contractId, ruleId }: { contractId: string; ruleId: string }) =>
    apiRequest('POST', `/api/contracts/${contractId}/rules/${ruleId}/approve`),
  onSuccess: () => {
    queryClient.invalidateQueries({ queryKey: ['/api/royalty-rules'] });
    toast({ title: 'Rule approved', description: 'Rule is now active in calculations' });
  },
});
 
const rejectMutation = useMutation({
  mutationFn: ({ contractId, ruleId, reason }: { contractId: string; ruleId: string; reason: string }) =>
    apiRequest('POST', `/api/contracts/${contractId}/rules/${ruleId}/reject`, { reason }),
  onSuccess: () => {
    queryClient.invalidateQueries({ queryKey: ['/api/royalty-rules'] });
    toast({ title: 'Rule rejected' });
  },
});

Critical edge case — what happens to contracts with all-pending rules
With the approval gate active, a contract whose rules are all in 'pending' status will return totalLicenseFee = 0 with an empty breakdown. This will be confusing without clear feedback. Add a check at the start of calculateWithUniversalEvaluator: if rules.length === 0 (after the approved filter), check if there are pending rules and if so return a specific error:
if (rules.length === 0) {

      eq(contractRules.contractId, contractId),
      eq(contractRules.isActive, true),
