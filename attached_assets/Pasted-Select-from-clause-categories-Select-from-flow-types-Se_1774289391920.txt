SELECT 
  -- Core Contract
  c.id AS contract_uuid,
  c.contract_number,
  c.display_name,
  c.status AS processing_status,
  c.contract_type,
  c.contract_subtype,
  c.effective_start,
  c.effective_end,
  c.counterparty_name,
  c.organization_name,
  c.company_id,
  -- Contract Master (structured metadata)
  cm.contract_id AS cm_contract_id,
  cm.contract_name,
  cm.contract_category,
  cm.contract_status,
  cm.territory_scope,
  cm.channel_scope,
  cm.owning_party,
  cm.counterparty_type,
  cm.contract_value_estimated_annual,
  cm.currency,
  cm.payment_frequency,
  cm.auto_renew,
  cm.renewal_term_months,
  cm.linked_contract_id,
  -- Company Master
  comp.company_name,
  comp.legal_entity_name,
  comp.industry,
  comp.headquarters_country,
  comp.annual_revenue_millions,
  comp.erp_system,
  comp.primary_currency,
  comp.fiscal_year_end,
  -- AI Analysis
  ca.summary,
  ca.key_terms,
  ca.confidence AS analysis_confidence,
  ca.header_review_flags,
  -- Counts
  (SELECT COUNT(*) FROM contract_terms ct WHERE ct.contract_id = cm.contract_id) AS term_count,
  (SELECT COUNT(*) FROM partner_contract_associations pca WHERE pca.contract_id = cm.contract_id) AS partner_count,
  (SELECT COUNT(*) FROM royalty_rules rr WHERE rr.contract_id = c.id AND rr.is_active = true) AS active_rules,
  (SELECT COUNT(*) FROM contract_clauses cc WHERE cc.contract_id = c.id) AS clause_count,
  (SELECT COUNT(*) FROM sales_data sd WHERE sd.matched_contract_id = c.id) AS sales_transactions
FROM contracts c
-- Contract Master (join on contract_number, no FK exists)
LEFT JOIN contract_master cm 
  ON cm.contract_number = c.contract_number
-- Company Master (join on company_id)
LEFT JOIN company_master comp 
  ON comp.company_id = c.company_id
-- AI Analysis
LEFT JOIN contract_analysis ca 
  ON ca.contract_id = c.id
ORDER BY c.created_at DESC;
